  window.NIRCloud = (() => {
    let client = null;
    let currentUser = null;
    let currentProfile = null;
    let saveTimer = null;
    let ready = false;
    let realtimeChannel = null;
    const stateKey = "nirCemetronFinalStateV2";

    const el = id => document.getElementById(id);

    function setStatus(text, kind="pending"){
      el("cloudStatusText").textContent = text;
      const dot = el("cloudStatusDot");
      dot.className = "cloud-status-dot";
      if(kind === "online") dot.classList.add("online");
      if(kind === "error") dot.classList.add("error");
    }

    function showAuthError(message){
      const box = el("cloudAuthError");
      box.textContent = message;
      box.style.display = "block";
    }

    function readRuntimeConfig(){
      try{
        const saved = JSON.parse(localStorage.getItem("nirSupabaseConfig") || "null");
        if(saved?.SUPABASE_URL && saved?.SUPABASE_ANON_KEY) return saved;
      }catch(_){}
      return window.NIR_CONFIG || {};
    }

    function configured(){
      const cfg = readRuntimeConfig();
      return cfg.SUPABASE_URL && cfg.SUPABASE_ANON_KEY &&
        !cfg.SUPABASE_URL.includes("SEU-PROJETO");
    }

    async function fetchProfile(){
      const {data, error} = await client
        .from("profiles")
        .select("id,nome,username,email,perfil,ativo")
        .eq("id", currentUser.id)
        .maybeSingle();
      if(error) throw error;
      if(!data || !data.ativo) throw new Error("Usuário sem perfil ativo.");
      currentProfile = data;
      el("cloudAdminButton").classList.toggle("hidden", data.perfil !== "ADMIN");
    }

    async function loadCloudState(){
      const {data, error} = await client
        .from("user_app_state")
        .select("state,updated_at")
        .eq("user_id", currentUser.id)
        .maybeSingle();
      if(error) throw error;

      const localRaw = localStorage.getItem(stateKey);
      const localState = localRaw ? JSON.parse(localRaw) : null;

      if(data?.state){
        state = data.state;
        localStorage.setItem(stateKey, JSON.stringify(state));
      } else if(localState){
        state = localState;
        await persistState(state);
      }

      state.records = Array.isArray(state.records) ? state.records : [];
      state.history = Array.isArray(state.history) ? state.history : [];
      state.calendarShifts = Array.isArray(state.calendarShifts) ? state.calendarShifts : [];
      state.sbar = state.sbar || {};
      renderAll();
      ensureCurrentRequestTime();
    }

    async function persistState(nextState){
      setStatus("Salvando...", "pending");
      const {error} = await client.from("user_app_state").upsert({
        user_id: currentUser.id,
        state: nextState,
        updated_at: new Date().toISOString()
      }, {onConflict:"user_id"});
      if(error){
        setStatus("Erro ao sincronizar", "error");
        throw error;
      }
      setStatus("Sincronizado", "online");
    }

    function scheduleSave(nextState){
      clearTimeout(saveTimer);
      setStatus("Alterações pendentes", "pending");
      const snapshot = JSON.parse(JSON.stringify(nextState));
      saveTimer = setTimeout(() => persistState(snapshot).catch(console.error), 500);
    }

    async function clearCloudState(){
      const {error} = await client.from("user_app_state").delete().eq("user_id", currentUser.id);
      if(error) throw error;
    }

    function subscribeRealtime(){
      if(realtimeChannel) client.removeChannel(realtimeChannel);
      realtimeChannel = client.channel(`nir-state-${currentUser.id}`)
        .on("postgres_changes", {
          event:"UPDATE", schema:"public", table:"user_app_state",
          filter:`user_id=eq.${currentUser.id}`
        }, payload => {
          if(payload.new?.state){
            state = payload.new.state;
            localStorage.setItem(stateKey, JSON.stringify(state));
            renderAll();
            setStatus("Atualizado em tempo real", "online");
          }
        }).subscribe();
    }

    async function afterLogin(user){
      currentUser = user;
      await fetchProfile();
      await loadCloudState();
      subscribeRealtime();
      ready = true;
      el("cloudAuthOverlay").style.display = "none";
      el("cloudToolbar").style.display = "flex";
      setStatus("Sincronizado", "online");
    }

    async function init(){
      if(!configured()){
        document.getElementById("cloudAuthOverlay").style.display = "none";
        document.getElementById("setupOverlay").classList.add("open");
        return;
      }

      const cfg = readRuntimeConfig();
      client = window.supabase.createClient(
        cfg.SUPABASE_URL,
        cfg.SUPABASE_ANON_KEY,
        {auth:{persistSession:true,autoRefreshToken:true,detectSessionInUrl:true}}
      );

      const {data:{session}} = await client.auth.getSession();
      if(session?.user){
        try{ await afterLogin(session.user); }
        catch(error){ showAuthError(error.message); await client.auth.signOut(); }
      }

      client.auth.onAuthStateChange(async (_event, session) => {
        if(session?.user && !ready){
          try{ await afterLogin(session.user); }
          catch(error){ showAuthError(error.message); }
        }
      });
    }

    function normalizeUsername(value){
      return String(value || "")
        .trim()
        .toLowerCase()
        .normalize("NFD")
        .replace(/[\u0300-\u036f]/g, "")
        .replace(/[^a-z0-9._-]/g, "");
    }

    function usernameToInternalEmail(username){
      const normalized = normalizeUsername(username);
      if(!normalized) throw new Error("Informe um nome de usuário válido.");
      const cfg = readRuntimeConfig();
      const domain = cfg.LOGIN_EMAIL_DOMAIN || "nircemetron.local";
      return `${normalized}@${domain}`;
    }

    async function login(username,password){
      const email = usernameToInternalEmail(username);
      const {data,error} = await client.auth.signInWithPassword({email,password});
      if(error) throw new Error("Usuário ou senha inválidos.");
      await afterLogin(data.user);
    }

    async function logout(){
      await client.auth.signOut();
      location.reload();
    }

    async function invokeAdmin(body){
      const {data,error} = await client.functions.invoke("admin-users", {body});
      if(error) throw error;
      if(data?.error) throw new Error(data.error);
      return data;
    }

    async function loadUsers(){
      const target = el("cloudUsersList");
      target.textContent = "Carregando...";
      try{
        const result = await invokeAdmin({action:"list"});
        target.innerHTML = (result.users || []).map(user => `
          

            
${escapeHtml(user.nome || "")}
@${escapeHtml(user.username || "")}

            
${escapeHtml(user.perfil || "USUARIO")}

            
${user.ativo ? "Ativo" : "Inativo"}

            
          
`).join("") || "
Nenhum usuário encontrado.
";
      }catch(error){
        target.innerHTML = `
Erro: ${escapeHtml(error.message)}
`;
      }
    }

    async function createUser(payload){
      await invokeAdmin({action:"create", ...payload});
      await loadUsers();
    }

    async function toggleUser(id, ativo){
      await invokeAdmin({action:"toggle", id, ativo});
      await loadUsers();
    }

    return {
      init, login, logout, scheduleSave, clearCloudState, loadUsers, createUser, toggleUser,
      isReady:() => ready, profile:() => currentProfile,
      getClient:() => client,
      reinitialize: async () => {
        ready = false;
        client = null;
        await init();
      }
    };
  })();

  document.getElementById("cloudLoginForm").addEventListener("submit", async event => {
    event.preventDefault();
    document.getElementById("cloudAuthError").style.display = "none";
    try{
      await window.NIRCloud.login(
        document.getElementById("cloudUsername").value.trim(),
        document.getElementById("cloudPassword").value
      );
    }catch(error){
      const box = document.getElementById("cloudAuthError");
      box.textContent = error.message || "Não foi possível entrar.";
      box.style.display = "block";
    }
  });

  document.getElementById("cloudLogoutButton").addEventListener("click", () => window.NIRCloud.logout());
  document.getElementById("cloudAdminButton").addEventListener("click", () => {
    document.getElementById("cloudAdminModal").classList.add("open");
    window.NIRCloud.loadUsers();
  });
  document.getElementById("cloudAdminClose").addEventListener("click", () => {
    document.getElementById("cloudAdminModal").classList.remove("open");
  });
  document.getElementById("cloudCreateUserForm").addEventListener("submit", async event => {
    event.preventDefault();
    try{
      await window.NIRCloud.createUser({
        nome:document.getElementById("adminNewName").value.trim(),
        username:document.getElementById("adminNewUsername").value.trim(),
        password:document.getElementById("adminNewPassword").value,
        perfil:document.getElementById("adminNewRole").value
      });
      event.target.reset();
      toast("Usuário cadastrado.");
    }catch(error){
      alert("Não foi possível cadastrar: " + error.message);
    }
  });


  const NIR_SETUP_SQL = "create extension if not exists pgcrypto;\n\ncreate table if not exists public.profiles (\n  id uuid primary key references auth.users(id) on delete cascade,\n  nome text not null,\n  username text not null unique,\n  email text not null unique,\n  perfil text not null default 'USUARIO' check (perfil in ('ADMIN','USUARIO')),\n  ativo boolean not null default true,\n  created_at timestamptz not null default now(),\n  updated_at timestamptz not null default now()\n);\n\ncreate table if not exists public.user_app_state (\n  user_id uuid primary key references public.profiles(id) on delete cascade,\n  state jsonb not null default '{}'::jsonb,\n  updated_at timestamptz not null default now()\n);\n\ncreate table if not exists public.audit_logs (\n  id bigint generated always as identity primary key,\n  actor_id uuid references public.profiles(id),\n  action text not null,\n  target_id text,\n  details jsonb,\n  created_at timestamptz not null default now()\n);\n\nalter table public.profiles enable row level security;\nalter table public.user_app_state enable row level security;\nalter table public.audit_logs enable row level security;\n\ncreate policy \"profiles_select_self_or_admin\"\non public.profiles for select to authenticated\nusing (\n  id = auth.uid()\n  or exists (\n    select 1 from public.profiles p\n    where p.id = auth.uid() and p.perfil = 'ADMIN' and p.ativo\n  )\n);\n\ncreate policy \"state_select_own\"\non public.user_app_state for select to authenticated\nusing (user_id = auth.uid());\n\ncreate policy \"state_insert_own\"\non public.user_app_state for insert to authenticated\nwith check (user_id = auth.uid());\n\ncreate policy \"state_update_own\"\non public.user_app_state for update to authenticated\nusing (user_id = auth.uid())\nwith check (user_id = auth.uid());\n\ncreate policy \"state_delete_own\"\non public.user_app_state for delete to authenticated\nusing (user_id = auth.uid());\n\ncreate policy \"audit_select_admin\"\non public.audit_logs for select to authenticated\nusing (\n  exists (\n    select 1 from public.profiles p\n    where p.id = auth.uid() and p.perfil = 'ADMIN' and p.ativo\n  )\n);\n\ncreate or replace function public.handle_new_user()\nreturns trigger\nlanguage plpgsql\nsecurity definer set search_path = public\nas $$\nbegin\n  insert into public.profiles (id, nome, username, email, perfil, ativo)\n  values (\n    new.id,\n    coalesce(new.raw_user_meta_data->>'nome', split_part(new.email,'@',1)),\n    coalesce(new.raw_user_meta_data->>'username', split_part(new.email,'@',1)),\n    new.email,\n    coalesce(new.raw_user_meta_data->>'perfil','USUARIO'),\n    true\n  )\n  on conflict (id) do nothing;\n  return new;\nend;\n$$;\n\ndrop trigger if exists on_auth_user_created on auth.users;\ncreate trigger on_auth_user_created\nafter insert on auth.users\nfor each row execute function public.handle_new_user();\n\nalter publication supabase_realtime add table public.user_app_state;\n";
  const setupEl = id => document.getElementById(id);
  setupEl("setupSqlPreview").textContent = NIR_SETUP_SQL;

  function setupMessage(id, message, ok){
    const node = setupEl(id);
    node.textContent = message;
    node.className = "setup-message " + (ok ? "ok" : "error");
  }

  setupEl("setupSaveConnection").addEventListener("click", async () => {
    const url = setupEl("setupUrl").value.trim().replace(/\/$/,"");
    const key = setupEl("setupKey").value.trim();
    if(!/^https:\/\/[a-z0-9-]+\.supabase\.co$/i.test(url) || !key){
      setupMessage("setupConnectionMessage","Informe a Project URL e a Publishable key completas.",false);
      return;
    }
    localStorage.setItem("nirSupabaseConfig", JSON.stringify({
      SUPABASE_URL:url, SUPABASE_ANON_KEY:key, LOGIN_EMAIL_DOMAIN:"nircemetron.local"
    }));
    try{
      const testClient = window.supabase.createClient(url,key);
      const {error} = await testClient.from("profiles").select("id").limit(1);
      if(error && (error.status === 401 || /invalid.*key|unauthorized/i.test(error.message))) throw error;
      setupMessage("setupConnectionMessage","Conexão salva. A chave pública foi aceita.",true);
      setupEl("setupStep1").classList.add("done");
    }catch(error){
      setupMessage("setupConnectionMessage","Falha na conexão: " + error.message,false);
    }
  });

  setupEl("setupCopySql").addEventListener("click", async () => {
    await navigator.clipboard.writeText(NIR_SETUP_SQL);
    setupMessage("setupSchemaMessage","Script copiado. Cole no SQL Editor do Supabase e execute.",true);
  });

  setupEl("setupOpenSql").addEventListener("click", () => {
    window.open("https://supabase.com/dashboard/project/cvtvtquyabfionisdfwg/sql/new","_blank","noopener");
  });

  setupEl("setupVerifySchema").addEventListener("click", async () => {
    try{
      const cfg = JSON.parse(localStorage.getItem("nirSupabaseConfig") || "null");
      if(!cfg) throw new Error("Salve primeiro a conexão.");
      const c = window.supabase.createClient(cfg.SUPABASE_URL,cfg.SUPABASE_ANON_KEY);
      const {error} = await c.from("profiles").select("id").limit(1);
      if(error) throw error;
      setupMessage("setupSchemaMessage","Banco verificado com sucesso.",true);
      setupEl("setupStep2").classList.add("done");
    }catch(error){
      setupMessage("setupSchemaMessage","O banco ainda não está pronto: " + error.message,false);
    }
  });

  setupEl("setupCreateAdmin").addEventListener("click", async () => {
    try{
      const cfg = JSON.parse(localStorage.getItem("nirSupabaseConfig") || "null");
      if(!cfg) throw new Error("Salve primeiro a conexão.");
      const username = setupEl("setupAdminUsername").value.trim().toLowerCase();
      const password = setupEl("setupAdminPassword").value;
      const nome = setupEl("setupAdminName").value.trim();
      if(!username || password.length < 8) throw new Error("Informe o usuário e uma senha com pelo menos 8 caracteres.");
      const c = window.supabase.createClient(cfg.SUPABASE_URL,cfg.SUPABASE_ANON_KEY);
      const email = `${username}@${cfg.LOGIN_EMAIL_DOMAIN || "nircemetron.local"}`;
      const {data,error} = await c.auth.signUp({
        email,password,options:{data:{nome,username,perfil:"ADMIN"}}
      });
      if(error && !/already registered/i.test(error.message)) throw error;
      if(data?.user){
        const {error:profileError} = await c.from("profiles").update({
          nome,username,perfil:"ADMIN",ativo:true
        }).eq("id",data.user.id);
        if(profileError) throw profileError;
      }
      setupMessage("setupAdminMessage","Administrador criado. Usuário: " + username,true);
      setupEl("setupStep3").classList.add("done");
    }catch(error){
      setupMessage("setupAdminMessage","Não foi possível criar o administrador: " + error.message,false);
    }
  });

  setupEl("setupFinish").addEventListener("click", async () => {
    if(!localStorage.getItem("nirSupabaseConfig")){
      setupMessage("setupConnectionMessage","Configure a conexão antes de concluir.",false);
      return;
    }
    setupEl("setupOverlay").classList.remove("open");
    document.getElementById("cloudAuthOverlay").style.display = "grid";
    await window.NIRCloud.reinitialize();
  });

  window.NIRCloud.init();
